public int[][] construct2DArray(int[] original, int m, int n) {
    int length = original.length;
    if (length != m * n) {
        return new int[0][0];
    }

    int[][] result = new int[m][n];
    for (int i = 0; i < length; i++) {
        result[i / n][i % n] = original[i];
    }

    return result;
}
